\begin{cases}
\dot x=5x + 2y + 1\\

\item   Consider the following bimatrix game:


\begin{tabular}{c|ccc}
 & D & E & F \\
\hline
A & 4;1 & 2;0 & 2;-1  \\
B & -2;6 & 4;5 & 2;2  \\
C & 1;0 & 3;-1 & 3;1  \\
\end{tabular}

\begin{enumerate}
\item Find all the pure  Nash equilibria
\item Find all the mixed Nash equilibria

\end{enumerate}
